def add(name, value, opts = {})
  cookie = {
    :name    => name,
    :value   => value,
    :secure  => opts[:secure],
    :path    => opts[:path],
    :expires => opts[:expires],
  }

  if opts[:domain]
    cookie[:domain] = opts[:domain]
  end

  @control.add_cookie cookie
end

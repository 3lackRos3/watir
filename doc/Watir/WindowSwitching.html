def window(*args, &blk)
  win = filter_windows(args, windows, :find)

  if win && block_given?
    win.use(&blk)
  end

  win
end

def windows(*args)
  wins = @driver.window_handles.map do |id|
    Window.new(@driver, id)
  end

  if args.empty?
    wins
  else
    filter_windows(args, wins, :select)
  end
end

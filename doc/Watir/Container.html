def extract_selector(selectors)
  case selectors.size
  when 2
    return { selectors[0] => selectors[1] }
  when 1
    obj = selectors.first
    return obj if obj.kind_of? Hash
  when 0
    return {}
  end

  raise ArgumentError, "expected Hash or (:how, 'what'), got #{selectors.inspect}"
end

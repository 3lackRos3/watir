def process
  attributes = try_eval

  if attributes.nil?
    p :ignoring => statement.source, :in => namespace.to_s
    return
  end

  TYPES.each do |type, return_type|
    if attributes.member? type
      create_attributes attributes[type], return_type
    end
  end
end
